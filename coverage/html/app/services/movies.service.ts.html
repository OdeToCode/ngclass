@Injectable()
export class MovieService {
 
    constructor(private http: Http) {
 
    }
 
    getAll() {
        return this.http.get("movies.json")
                   .map(res => res.json().map(o => 
                        new Movie(o.id, o.title, o.rating, o.length)));
    }
    
}
